def analisar_qualidade_amostra(limite: int = 100) -> dict:
  """Analisa completude dos campos obrigatórios
  em uma amostra de consumidores."""
  query = f"""
    SELECT COUNT(*) AS TOTAL,
    SUM(CASE WHEN MEDIDOR_ID IS NULL
        THEN 1 ELSE 0 END) AS SEM_MEDIDOR
    FROM TAB_CADASTRO
    LIMIT {limite}
  """
  df = read_sql_to_dataframe(query)
  return df.iloc[0].to_dict()
