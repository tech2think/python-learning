CTEs — organização de queries

query = """
WITH nulos_por_campo AS (
  SELECT
    COUNT(*) AS total,
    COUNT(cpf_cnpj) AS com_cpf,
    COUNT(num_medidor) AS com_medidor,
    COUNT(num_cep) AS com_cep
  FROM CONSUMIDORES_UC
  WHERE flg_ativo = TRUE
),
pct_preenchimento AS (
  SELECT
    ROUND(com_cpf / total * 100, 1)
      AS pct_cpf,
    ROUND(com_medidor / total * 100, 1)
      AS pct_medidor
  FROM nulos_por_campo
)
SELECT * FROM pct_preenchimento;
"""

Window Functions — ranking de issues

query_rank = """
SELECT
  nom_municipio,
  COUNT(*) AS total_ucs,
  SUM(CASE WHEN cpf_cnpj IS NULL
    THEN 1 ELSE 0 END) AS sem_cpf,
  ROUND(
    SUM(CASE WHEN cpf_cnpj IS NULL
      THEN 1 ELSE 0 END)
    / COUNT(*) * 100, 1
  ) AS pct_sem_cpf,
  RANK() OVER (
    ORDER BY pct_sem_cpf DESC
  ) AS ranking
FROM CONSUMIDORES_UC
GROUP BY nom_municipio
ORDER BY ranking
LIMIT 20;
"""
