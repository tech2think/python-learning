M3 Visualização de Indicadores

matplotlib — Barras

import matplotlib.pyplot as plt

classes = df.groupby(
  'cod_classe_consumo'
)['cod_consumidor'].count()

classes.plot(kind='bar',
  color='steelblue',
  title='Consumidores
  por Classe')
plt.tight_layout()
plt.show()

seaborn — Heatmap

import seaborn as sns

nulos = df.isnull().groupby(
  df['cod_uf']
).mean() * 100

sns.heatmap(
  nulos,
  annot=True,
  fmt='.0f',
  cmap='RdYlGn_r'
)
plt.title('% Nulos / UF')

plotly — Interativo

import plotly.express as px

fig = px.line(
  df_mensal,
  x='mes',
  y='consumo_kwh',
  color='classe',
  title='Consumo Mensal'
)
fig.update_layout(
  template='plotly_white'
)
fig.show()
