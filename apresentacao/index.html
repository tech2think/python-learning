def classificar_tarifa(consumo_kwh):
  if consumo_kwh <= 30:
    return "Baixa Renda"
  elif consumo_kwh <= 100:
    return "Faixa 1"
  elif consumo_kwh <= 220:
    return "Faixa 2"
  else:
    return "Faixa 3"

M1 Funções

Cálculo do DEC

def calcular_dec(interrupcoes: list) -> float:
  """
  Calcula o DEC (Duração Equivalente de
  Interrupção por Consumidor).

  Args:
    interrupcoes: lista de dicts com
      'duracao_h' e 'consumidores'
  Returns:
    DEC em horas
  """
  total_ucs = interrupcoes[0]['total_ucs']
  soma = sum(
    i['duracao_h'] * i['consumidores']
    for i in interrupcoes
  )
  return round(soma / total_ucs, 4)

Validação de CPF

import re

def validar_cpf_formato(cpf: str) -> bool:
  """Valida formato do CPF."""
  cpf_limpo = re.sub(r'\D', '', cpf)
  return len(cpf_limpo) == 11

def formatar_cpf(cpf: str) -> str:
  """Formata CPF: 12345678900 →
  123.456.789-00"""
  c = re.sub(r'\D', '', cpf)
  if len(c) != 11:
    return cpf
  return f"{c[:3]}.{c[3:6]}.{c[6:9]}-{c[9:]}"

M2 Limpeza de Dados de Consumidores

Problemas comuns no cadastro

  • CPF/CNPJ com formatação inconsistente
  • Nomes em maiúsculas, minúsculas e misto
  • CEP sem zeros à esquerda
  • Duplicatas de UC por migração de sistema
  • Datas em formatos diferentes
  • Campos obrigatórios vazios
  • Coordenadas geográficas ausentes
Dados de baixa qualidade impactam diretamente o faturamento e os indicadores ANEEL.

Técnicas de limpeza

# Normalizar nomes
df['nom_consumidor'] = (
  df['nom_consumidor']
  .str.strip()
  .str.title()
)

# Limpar CPF/CNPJ
df['cpf_cnpj'] = (
  df['cpf_cnpj']
  .str.replace(r'\D', '', regex=True)
)

# Remover duplicatas
df = df.drop_duplicates(
  subset=['cod_consumidor']
)

M2 Análise de Qualidade do Cadastro

Score de qualidade por registro

def score_qualidade(row) -> float:
  campos = [
    'nom_consumidor', 'cpf_cnpj',
    'cod_modalidade_tarifaria',
    'nom_municipio', 'num_cep',
    'num_medidor', 'dat_ultima_leitura'
  ]
  preenchidos = sum(
    1 for c in campos
    if pd.notna(row[c]) and row[c] != ''
  )
  return round(preenchidos / len(campos), 2)

df['score_qualidade'] = df.apply(
  score_qualidade, axis=1
)

M3 Indicadores DEC e FEC

DEC — Duração Equivalente de Interrupção

Soma das durações de interrupções ponderadas pelo número de UCs afetadas, dividida pelo total de UCs do conjunto.

# Fórmula
DEC = Σ(Duração_i × UCs_i) / UCs_total

FEC — Frequência Equivalente de Interrupção

Número de interrupções ponderadas pelo número de UCs afetadas, dividido pelo total de UCs.

FEC = Σ(UCs_i) / UCs_total
